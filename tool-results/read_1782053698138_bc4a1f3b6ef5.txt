   131→function fmt(n: number): string {
   132→  if (!Number.isFinite(n)) return '0';
   133→  if (n >= 1_000_000) return `${(n / 1_000_000).toFixed(2)}M`;
   134→  if (n >= 1_000) return `${(n / 1_000).toFixed(1)}k`;
   135→  return `${Math.round(n)}`;
   136→}
   137→
   138→function ago(ts: number | undefined): string {
   139→  if (!ts) return 'never';
   140→  const diff = Date.now() - ts;
   141→  if (diff < 0) return 'just now';
   142→  if (diff < 60_000) return `${Math.round(diff / 1000)}s ago`;
   143→  if (diff < 3_600_000) return `${Math.round(diff / 60_000)}m ago`;
   144→  return `${Math.round(diff / 3_600_000)}h ago`;
   145→}
   146→
   147→function humanizeReason(reason: string): string {
   148→  if (!reason) return '—';
   149→  return reason
   150→    .replace(/-/g, ' ')
   151→    .replace(/\b\w/g, (c) => c.toUpperCase());
   152→}
   153→
   154→function actionColor(action: AssetWatch['action']): string {
   155→  switch (action) {
   156→    case 'analyze':
   157→      return 'text-sky-300 bg-sky-500/15';
   158→    case 'cache':
   159→      return 'text-violet-300 bg-violet-500/15';
   160→    case 'skip':
   161→      return 'text-muted-foreground bg-muted/40';
   162→  }
   163→  return 'text-muted-foreground bg-muted/40';
   164→}
   165→
   166→function tierLabel(tier?: number): string {
   167→  if (tier === 1) return 'T1 · triage';
   168→  if (tier === 2) return 'T2 · deep';
   169→  return '—';
   170→}
   171→
   172→function actionIcon(action: AssetWatch['action']): React.ReactNode {
   173→  switch (action) {
   174→    case 'analyze':
   175→      return <Zap className="h-3 w-3 text-sky-400" />;
   176→    case 'cache':
   177→      return <DatabaseZap className="h-3 w-3 text-violet-300" />;
   178→    case 'skip':
   179→      return <Eye className="h-3 w-3 text-muted-foreground" />;
   180→  }
   181→  return <Eye className="h-3 w-3 text-muted-foreground" />;
   182→}

   184→function regimeBadge(regime: AssetWatch['regime']): React.ReactNode {
   185→  const map: Record<AssetWatch['regime'], { variant: 'success' | 'warning' | 'destructive'; label: string }> = {
   186→    trending: { variant: 'success', label: 'trend' },
   187→    ranging: { variant: 'warning', label: 'range' },
   188→    volatile: { variant: 'destructive', label: 'vol' },
   189→  };
   190→  const m = map[regime];
   191→  return <Badge variant={m.variant} className="text-[9px]">{m.label}</Badge>;
   192→}

   201→  min: number;

   209→function CfgSlider({
   210→  label,
   211→  value,
   212→  min,
   213→  max,
   214→  step,
   215→  unit,
   216→  description,
   217→  onChange,
   218→}: CfgSliderProps): React.ReactElement {

   220→    <div className="space-y-2">

   222→        <span className="text-[11px] font-medium text-foreground">{label}</span>

   228→      <Slider
   229→        value={[value]}
   230→        min={min}
   231→        max={max}
   232→        step={step}
   233→        onValueChange={(v) => onChange(v[0])}
   234→      />
   235→      {description ? (
   236→        <p className="text-[10px] text-muted-foreground/70">{description}</p>
   237→      ) : null}
   238→    </div>
   239→  );
   240→}

   249→  const brainQ = useQuery<BrainSnapshot>({
   250→    queryKey: ['brain'],

   259→    staleTime: 3000,
   260→  });

   266→      if (!res.ok) throw new Error('analytics fetch failed');
   267→      const json: { success?: boolean; data?: AnalyticsModelsResponse } = await res.json();
   268→      return json.data ?? {};

   270→    refetchInterval: 60_000,
   271→    staleTime: 45_000,
   272→  });
   273→
   274→  const controlMut = useMutation({
   275→    mutationFn: async (payload: Record<string, unknown>) => {
   276→      const res = await fetch('/api/brain', {
   277→        method: 'POST',
   278→        headers: { 'Content-Type': 'application/json' },
   279→        body: JSON.stringify(payload),
   280→      });
   281→      if (!res.ok) throw new Error('brain control failed');
   282→      return res.json();
   283→    },
   284→    onSuccess: () => {
   285→      qc.invalidateQueries({ queryKey: ['brain'] });
   286→    },
   287→  });
   288→
   289→  const b = brainQ.data;
   290→  const samples = b?.statsSamples ?? [];
   291→  const stats = b?.stats;
   292→  const watch = b?.watch ?? [];
   293→  const actions = b?.recentActions ?? [];
   294→  const tuneEvents = b?.tuneEvents ?? [];
   295→  const config = b?.config;
   296→
   297→  const tokensUsed = stats?.tokensUsed ?? 0;
   298→  const tokensSaved = stats?.tokensSaved ?? 0;
   299→  const gross = tokensUsed + tokensSaved;
   300→  const savedPct = gross > 0 ? (tokensSaved / gross) * 100 : 0;
   301→  const triggersTotal =
   302→    (stats?.triggersNews ?? 0) +
   303→    (stats?.triggersCrossAsset ?? 0) +
   304→    (stats?.triggersManual ?? 0);
   305→
   306→  const winRate = analyticsQ.data?.overall?.overallAccuracy ?? 0;
   307→

   309→  // the current watch list and queue each symbol as a manual force-run.
   310→  const forceRunAll = React.useCallback(() => {
   311→    if (!watch.length) return;
   312→    for (const w of watch) {
   313→      controlMut.mutate({ action: 'forceRun', symbol: w.symbol });
   314→    }
   315→  }, [watch, controlMut]);
   316→
   317→  const isRunning = !!b?.running;
   318→  const inCooldown = !!b?.llmInCooldown;
   319→  const isManual = b?.mode === 'manual';
   320→
   321→  const triggerSegments: TriggerSegment[] = [
   322→    { key: 'news', label: 'News', count: stats?.triggersNews ?? 0, color: '' },
   323→    { key: 'cross-asset', label: 'Cross-asset', count: stats?.triggersCrossAsset ?? 0, color: '' },
   324→    { key: 'manual', label: 'Manual', count: stats?.triggersManual ?? 0, color: '' },
   325→  ];
   326→
   327→  const budgetPct = b && b.budgetCap > 0 ? (b.budgetUsed / b.budgetCap) * 100 : 0;
   328→
   329→  return (
   330→    <div className="space-y-4 p-4 md:p-6">
   331→      {/* ---------- Header ---------- */}
   332→      <div className="relative overflow-hidden rounded-xl border border-border bg-gradient-to-br from-violet-950/40 via-card to-card p-5">
   333→        <div
   334→          className="absolute -right-12 -top-12 h-48 w-48 rounded-full bg-violet-500/20 blur-3xl ambient-glow"
   335→          aria-hidden
   336→        />
   337→        <div className="relative flex flex-wrap items-center justify-between gap-4">
   338→          <div className="flex items-center gap-3">
   339→            <div className="relative flex h-11 w-11 items-center justify-center rounded-xl bg-violet-500/15">
   340→              <BrainCircuit className="h-5 w-5 text-violet-300" />
   341→              <span
   342→                className={cn(
   343→                  'absolute -right-0.5 -top-0.5 flex h-3 w-3',
   344→                )}
   345→              >
   346→                <span
   347→                  className={cn(
   348→                    'absolute inline-flex h-full w-full animate-ping rounded-full opacity-70',
   349→                    !isRunning ? 'bg-rose-500' : inCooldown ? 'bg-amber-400' : 'bg-emerald-400',
   350→                  )}
   351→                />
   352→                <span
   353→                  className={cn(
   354→                    'relative inline-flex h-3 w-3 rounded-full',
   355→                    !isRunning ? 'bg-rose-500' : inCooldown ? 'bg-amber-400' : 'bg-emerald-400',
   356→                  )}
   357→                />
   358→              </span>
   359→            </div>
   360→            <div className="flex flex-col gap-1">
   361→              <h1 className="text-gradient bg-gradient-to-r from-violet-300 via-fuchsia-300 to-emerald-300 text-2xl font-bold tracking-tight">
   362→                The Lazy Brain
   363→              </h1>
   364→              <div className="flex items-center gap-2">
   365→                <motion.div
   366→                  animate={{
   367→                    boxShadow: isRunning
   368→                      ? [
   369→                          '0 0 0 0 oklch(0.72 0.18 160 / 0.6)',
   370→                          '0 0 0 6px oklch(0.72 0.18 160 / 0)',
   371→                        ]
   372→                      : '0 0 0 0 oklch(0.65 0.22 25 / 0.6)',
   373→                  }}
   374→                  transition={{ duration: 1.6, repeat: Infinity, ease: 'easeOut' }}
   375→                  className={cn(
   376→                    'inline-flex items-center gap-1.5 rounded-full border px-2 py-0.5 text-[10px] font-bold uppercase tracking-wider',
   377→                    !isRunning
   378→                      ? 'border-rose-500/40 bg-rose-500/10 text-rose-300'
   379→                      : inCooldown
   380→                        ? 'border-amber-500/40 bg-amber-500/10 text-amber-300'
   381→                        : 'border-emerald-500/40 bg-emerald-500/10 text-emerald-300',
   382→                  )}
   383→                >
   384→                  <span
   385→                    className={cn(
   386→                      'h-1.5 w-1.5 rounded-full',
   387→                      !isRunning ? 'bg-rose-400' : inCooldown ? 'bg-amber-400' : 'bg-emerald-400',
   388→                    )}
   389→                  />
   390→                  {isRunning ? (inCooldown ? 'Cooldown' : 'Autonomous') : 'Paused'}
   391→                </motion.div>
   392→                <Badge variant={isManual ? 'warning' : 'info'} className="text-[10px]">
   393→                  {isManual ? 'manual' : 'auto'}
   394→                </Badge>
   395→                <ThinkingIndicator />
   396→              </div>
   397→            </div>
   398→          </div>
   399→        </div>
   400→      </div>
   401→
   402→      {/* ---------- LLM cooldown banner ---------- */}
   403→      {inCooldown ? (
   404→        <div className="flex items-center gap-2 rounded-lg border border-amber-500/40 bg-amber-500/10 px-4 py-2 text-xs text-amber-200">
   405→          <Ban className="h-3.5 w-3.5" />
   406→          <span>
   407→            LLM circuit-breaker tripped — cooldown until{' '}
   408→            <span className="font-mono">

   413→        </div>

   415→

   856→function ScoreTile({
   857→  label,
   858→  value,
   859→  sub,
   860→  icon,
   861→  accent,
   862→}: {
   863→  label: string;
   864→  value: string;
   865→  sub?: string;
   866→  icon: React.ReactNode;
   867→  accent: 'amber' | 'emerald' | 'violet' | 'fuchsia' | 'rose' | 'muted';
   868→}): React.ReactElement {
   869→  const accentMap: Record<typeof accent, string> = {
   870→    amber: 'border-amber-500/30 bg-amber-500/5',
   871→    emerald: 'border-emerald-500/30 bg-emerald-500/5',
   872→    violet: 'border-violet-500/30 bg-violet-500/5',
   873→    fuchsia: 'border-fuchsia-500/30 bg-fuchsia-500/5',
   874→    rose: 'border-rose-500/30 bg-rose-500/5',
   875→    muted: 'border-border bg-muted/20',
   876→  };
   877→  return (
   878→    <div className={cn('rounded-lg border p-3', accentMap[accent])}>
   879→      <div className="flex items-center justify-between">
   880→        <span className="text-[10px] uppercase tracking-wider text-muted-foreground">
   881→          {label}
   882→        </span>
   883→        {icon}
   884→      </div>
   885→      <div className="mt-1 text-xl font-bold text-foreground">{value}</div>
   886→      {sub ? <div className="mt-0.5 text-[10px] text-muted-foreground">{sub}</div> : null}
   887→    </div>
   888→  );
   889→}
